150 {
151  // initiate a new conversion
152  initialize_memory_write(0);
153  spi_write(0b10000000 | channel_number); // start a conversion
154  finish_memory_write();
155 
156  while (conversion_done()==0); // wait for conversion to complete
157 }

203 float convert_to_signed_float(unsigned char results[4]) {
204  // Get the last 24 bits of the results (the first 8 bits are status bits)
205  long x = 0L;
206  x= x | ((unsigned long) results[2]<<16)
207  | ((unsigned long) results[1]<<8)
208  | ((unsigned long) results[0]);
209 
210  // Convert a 24-bit two's complement number into a 32-bit two's complement number
211  boolean sign;
212  if ((results[2]&0b10000000)==128) sign=true; else sign=false;
213  if (sign) x=x | 0xFF000000;
214 
215  return float(x);
216 }

242 // Translate the fault byte into usable fault data and print it out
243 void print_fault_data(unsigned char fault_byte) {
244  Serial.print(" FAULT DATA=");
245  Serial.println(fault_byte,BIN);
246 
247  if ((fault_byte&SENSOR_HARD_FAILURE)>0) Serial.println("*SENSOR HARD FALURE*");
248  if ((fault_byte&ADC_HARD_FAILURE)>0) Serial.println("*ADC_HARD_FAILURE*");
249  if ((fault_byte&CJ_HARD_FAILURE)>0) Serial.println("*CJ_HARD_FAILURE*");
250  if ((fault_byte&CJ_SOFT_FAILURE)>0) Serial.println("*CJ_SOFT_FAILURE*");
251  if ((fault_byte&SENSOR_ABOVE)>0) Serial.println("*SENSOR_ABOVE*");
252  if ((fault_byte&SENSOR_BELOW)>0) Serial.println("*SENSOR_BELOW*");
253  if ((fault_byte&ADC_RANGE_ERROR)>0) Serial.println("*ADC_RANGE_ERROR*");
254  if ((fault_byte&VALID)!=1) Serial.println("!!!!!!! INVALID READING !!!!!!!!!");
255  if (fault_byte==0b11111111) Serial.println("&&&&&&&&&& CONFIGURATION ERROR &&&&&&&&&&&&");
256  Serial.println(" ");
257 }
