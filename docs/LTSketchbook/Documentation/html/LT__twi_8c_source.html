60 /*
61  * Function twi_init
62  * Desc readys twi pins and sets twi bitrate
63  * Input none
64  * Output none
65  */
66 void twi_init(void)
67 {
68  twi_commBuffer = 0;
69 
70  // initialize state
71  twi_state = TWI_READY;
72  twi_sendStop = true; // default value
73  twi_inRepStart = false;
74 
75  // activate internal pullups for twi.
76  digitalWrite(SDA, 1);
77  digitalWrite(SCL, 1);
78 
79  // initialize twi prescaler and bit rate
80  cbi(TWSR, TWPS0);
81  cbi(TWSR, TWPS1);
82  TWBR = ((F_CPU / TWI_FREQ) - 16) / 2;
83 
84  /* twi bit rate formula from atmega128 manual pg 204
85  SCL Frequency = CPU Clock Frequency / (16 + (2 * TWBR))
86  note: TWBR should be 10 or higher for master mode
87  It is 72 for a 16mhz Wiring board with 100kHz TWI */
88 
89  // enable twi module, acks, and twi interrupt
90  TWCR = _BV(TWEN) | _BV(TWIE) | _BV(TWEA);
91 }

95 /*
96  * Function twi_readFrom
97  * Desc attempts to become twi bus master and read a
98  * series of bytes from a device on the bus
99  * Input address: 7bit i2c device address
100  * data: pointer to byte array
101  * length: number of bytes to read into array
102  * sendStop: Boolean indicating whether to send a stop at the end
103  * Output number of bytes read
104  */
105 uint16_t twi_readFrom(uint8_t address, uint8_t* data, uint16_t length, uint8_t sendStop)
106 {
107  uint16_t i;
108 
109 
110  // wait until twi is ready, become master receiver
111  while(TWI_READY != twi_state){
112  continue;
113  }
114  twi_state = TWI_MRX;
115  twi_sendStop = sendStop;
116  // reset error state (0xFF.. no error occured)
117  twi_error = 0xFF;
118 
119  // initialize buffer iteration vars
120  twi_commBuffer = data;
121  twi_commBufferIndex = 0;
122  twi_commBufferLength = length-1; // This is not intuitive, read on...
123  // On receive, the previously configured ACK/NACK setting is transmitted in
124  // response to the received byte before the interrupt is signalled.
125  // Therefor we must actually set NACK when the _next_ to last byte is
126  // received, causing that NACK to be sent in response to receiving the last
127  // expected byte of data.
128 
129  // build sla+w, slave device address + w bit
130  twi_slarw = TW_READ;
131  twi_slarw |= address << 1;
132 
133  if (true == twi_inRepStart) {
134  // if we're in the repeated start state, then we've already sent the start,
135  // (@@@ we hope), and the TWI statemachine is just waiting for the address byte.
136  // We need to remove ourselves from the repeated start state before we enable interrupts,
137  // since the ISR is ASYNC, and we could get confused if we hit the ISR before cleaning
138  // up. Also, don't enable the START interrupt. There may be one pending from the
139  // repeated start that we sent outselves, and that would really confuse things.
140  twi_inRepStart = false; // remember, we're dealing with an ASYNC ISR
141  do
142  {
143  TWDR = twi_slarw;
144  } while (TWCR & _BV(TWWC));
145  TWCR = _BV(TWINT) | _BV(TWEA) | _BV(TWEN) | _BV(TWIE); // enable INTs, but not START
146  }
147  else
148  // send start condition
149  TWCR = _BV(TWEN) | _BV(TWIE) | _BV(TWEA) | _BV(TWINT) | _BV(TWSTA);
150 
151  // wait for read operation to complete
152  while(TWI_MRX == twi_state){
153  continue;
154  }
155 
156  if (twi_commBufferIndex < length)
157  length = twi_commBufferIndex;
158 
159 
160  return length;
161 }

165  * Function twi_writeTo
166  * Desc attempts to become twi bus master and write a
167  * series of bytes to a device on the bus
168  * Input address: 7bit i2c device address
169  * data: pointer to byte array
170  * length: number of bytes in array
171  * wait: boolean indicating to wait for write or not
172  * sendStop: boolean indicating whether or not to send a stop at the end
173  * Output 0 .. success
174  * 1 .. length to long for buffer
175  * 2 .. address send, NACK received
176  * 3 .. data send, NACK received
177  * 4 .. other twi error (lost bus arbitration, bus error, ..)

264 void twi_stop(void)
265 {
266  // send stop condition
267  TWCR = _BV(TWEN) | _BV(TWIE) | _BV(TWEA) | _BV(TWINT) | _BV(TWSTO);
268 
269  // wait for stop condition to be exectued on bus

272  continue;
273  }
274 
275  // update twi state

294 ISR(TWI_vect)
295 {
296  switch(TW_STATUS){
297  // All Master
298  case TW_START: // sent start condition
299  case TW_REP_START: // sent repeated start condition
300  // copy device address and r/w bit to output register and ack
301  TWDR = twi_slarw;
302  twi_reply(1);
303  break;
304 
305  // Master Transmitter
306  case TW_MT_SLA_ACK: // slave receiver acked address
307  case TW_MT_DATA_ACK: // slave receiver acked data
308  // if there is data to send, send it, otherwise stop
309  if(twi_commBufferIndex < twi_commBufferLength){
310  // copy data to output register and ack
311  TWDR = twi_commBuffer[twi_commBufferIndex++];
312  twi_reply(1);
313  }else{
314  if (twi_sendStop)
315  twi_stop();
316  else {
317  twi_inRepStart = true; // we're gonna send the START
318  // don't enable the interrupt. We'll generate the start, but we
319  // avoid handling the interrupt until we're in the next transaction,
320  // at the point where we would normally issue the start.
321  TWCR = _BV(TWINT) | _BV(TWSTA)| _BV(TWEN) ;
322  twi_state = TWI_READY;
323  }
324  }
325  break;
326  case TW_MT_SLA_NACK: // address sent, nack received
327  twi_error = TW_MT_SLA_NACK;
328  twi_stop();
329  break;
330  case TW_MT_DATA_NACK: // data sent, nack received
331  twi_error = TW_MT_DATA_NACK;
332  twi_stop();
333  break;
334  case TW_MT_ARB_LOST: // lost bus arbitration
335  twi_error = TW_MT_ARB_LOST;
336  twi_releaseBus();
337  break;
338 
339  // Master Receiver
340  case TW_MR_DATA_ACK: // data received, ack sent
341  // put byte into buffer
342  twi_commBuffer[twi_commBufferIndex++] = TWDR;
343  case TW_MR_SLA_ACK: // address sent, ack received
344  // ack if more bytes are expected, otherwise nack
345  if(twi_commBufferIndex < twi_commBufferLength){
346  twi_reply(1);
347  }else{
348  twi_reply(0);
349  }
350  break;
351  case TW_MR_DATA_NACK: // data received, nack sent
352  // put final byte into buffer
353  twi_commBuffer[twi_commBufferIndex++] = TWDR;
354  if (twi_sendStop)
355  twi_stop();
356  else {
357  twi_inRepStart = true; // we're gonna send the START
358  // don't enable the interrupt. We'll generate the start, but we
359  // avoid handling the interrupt until we're in the next transaction,
360  // at the point where we would normally issue the start.
361  TWCR = _BV(TWINT) | _BV(TWSTA)| _BV(TWEN) ;
362  twi_state = TWI_READY;
363  }
364  break;
365  case TW_MR_SLA_NACK: // address sent, nack received
366  twi_stop();
367  break;
368  // TW_MR_ARB_LOST handled by TW_MT_ARB_LOST case
369 
370 
371  // All
372  case TW_NO_INFO: // no state information
373  break;
374  case TW_BUS_ERROR: // bus error, illegal stop/start
375  twi_error = TW_BUS_ERROR;
376  twi_stop();
377  break;
378  }
379 }
