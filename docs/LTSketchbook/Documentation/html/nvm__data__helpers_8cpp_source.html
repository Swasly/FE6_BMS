49 /********************************************************************
50  * Function: nvramNode_p nvramListNew(uint16_t dataIn, uint8_t pecIn, uint8_t addIn, uint8_t cmdIn);
51  *
52  * PreCondition: None
53  * Input: uint16_t dataIn = WORD of data from NVM block
54  * uint8_t pecIn = Optionally use PEC or not, same as from t_RECORD_NVM_DATA->detailedRecordHeader->UsePec;
55  * uint8_t addIn = Address of device this block belongs to, same as from t_RECORD_NVM_DATA->detailedRecordHeader->DeviceAddress
56  * uint8_t cmdIn = Command for NVM block store of device, same as from t_RECORD_NVM_DATA->detailedRecordHeader->CommandCode
57  * Output: Returns a pointer to the newly created node on the stack/heap (depending on malloc implementation)
58  * Overview: Makes a new node for the Linked List
59  * Note: Note that depending on your implementation, you will need to closely
60  * monitor your stack or heap to ensure you have sufficient space to
61  * store all NVRAM data in this dynamically allocated linked list
62  * Use of malloc is required so that the node is not destroyed when the function returns
63  *******************************************************************/
64 nvramNode_p nvramListNew(uint16_t dataIn, uint8_t pecIn, uint8_t addIn, uint8_t cmdIn)
65 {
66  nvramNode_p newNode = (nvramNode_p)malloc(sizeof(nvramNode_t));
67  newNode->data_store = dataIn;
68  newNode->nvram_usePEC = pecIn;
69  newNode->nvram_pmbusAddress = addIn;
70  newNode->nvram_pmbusCommand = cmdIn;
71  newNode->next = NULL;
72  return newNode;
73 }

77  *
78  * PreCondition: Must have a typeof(nvramNode_t) list available to add to and store NVM data contents into
79  * Input: uint16_t dataIn = WORD of data from NVM block
80  * uint8_t pecIn = Optionally use PEC or not, same as from t_RECORD_NVM_DATA->detailedRecordHeader->UsePec;
81  * uint8_t addIn = Address of device this block belongs to, same as from t_RECORD_NVM_DATA->detailedRecordHeader->DeviceAddress
82  * uint8_t cmdIn = Command for NVM block store of device, same as from t_RECORD_NVM_DATA->detailedRecordHeader->CommandCode
83  * nvramNode_t* nvramList = Base address for the linked list to store to (Making this a parameter allows for multiple NVM lists in the future)
84  * Output: None
85  * Overview: Adds a new data point to the master NVRAM buffer list
86  * Note: First it navigates to the end node of the linked list, then creates a new node, and links it
87  * Note that to save processing power, you could save the last linked list address and reference that directly instead of finding it dynamically each time
88  *******************************************************************/
89 void nvramListAdd(uint16_t dataIn, uint8_t pecIn, uint8_t addIn, uint8_t cmdIn, nvramNode_t* nvramList)
90 {
91  nvramNode_p newNode;
92  nvramNode_p node = nvramList;
93 
94  while (node->next != NULL)
95  {
96  node = node->next;
97  }
98 
99  newNode = nvramListNew(dataIn, pecIn, addIn, cmdIn);
100  node->next = newNode;
101 }
