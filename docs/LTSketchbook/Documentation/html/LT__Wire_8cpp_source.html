118 //
119 // Originally, 'endTransmission' was an f(void) function.
120 // It has been modified to take one parameter indicating
121 // whether or not a STOP should be performed on the bus.
122 // Calling endTransmission(false) allows a sketch to
123 // perform a repeated start.
124 //
125 // WARNING: Nothing in the library keeps track of whether
126 // the bus tenure has been properly ended with a STOP. It
127 // is very possible to leave the bus in a hung state if
128 // no call to endTransmission(true) is made. Some I2C
129 // devices will behave oddly if they do not see a STOP.
130 //
131 uint8_t TwoWire::endTransmission(uint8_t sendStop)
132 {
133  // transmit buffer (blocking)
134  int8_t ret = twi_writeTo(txAddress, commBuffer, commBufferLength, 1, sendStop);
135 
136  free(commBuffer);
137  commBuffer = 0;
138  commBufferTxMax = 0;
139  // reset tx buffer iterator vars
140  commBufferIndex = 0;
141  commBufferLength = 0;
142  // indicate that we are done transmitting
143  transmitting = 0;
144  return ret;
145 }
